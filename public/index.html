Installation

        git clone <repository-url>
        npm install
        # Configure environment variables creating a .env file in the root directory and add this environment variables.
        DB_HOST=localhost
        DB_PORT=5432
        DB_USER=myuser
        DB_PASSWORD=mypassword
        npm start

Deployment

        npm run deploy
